\itemwhen \((\pi : E \to B)\) : fiber bundle over manifold with fiber manifold

  define \(\mathcal{V}E\) : sub bundle of \(TE\) := \(\coprod_{p \in E} \{X \in T_p E \mid X(\pi^* f) = 0\}\)
  define vertical vector fields := \(\Gamma(\mathcal{V}E)\)
\end{Definition}
